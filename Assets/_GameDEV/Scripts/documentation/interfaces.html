💡 Genel Tasarım Prensipleri

  • SOLID Prensipleri:
    • Interface Segregation Principle (ISP) - Küçük ve özel amaçlı arayüzler
    • Dependency Inversion Principle (DIP) - Yüksek seviye modüller arayüzlere bağlı
    • Single Responsibility Principle (SRP) - Her arayüz tek bir sorumluluğa sahip
  • Tasarım Desenleri:
    • Strategy Pattern - IProjectileStrategy ile mermi davranışları
    • Dependency Injection - PlayerController'da arayüz kullanımı
    • Interface-based Programming - Tüm sistemler arası iletişim

🎮 Kullanım Örneği


// PlayerController'da arayüzlerin kullanımı
public class PlayerController : MonoBehaviour
{
    [SerializeField] private IMoveable _movement;
    [SerializeField] private IGroundChecker _groundChecker;
    [SerializeField] private IMaterialScroller _materialScroller;
    [SerializeField] private IShooter _shooter;
    private IInputHandler[] _inputHandlers;

    private void Update()
    {
        Vector3 input = GetActiveInput();
        if (_groundChecker.IsGrounded)
        {
            _movement.Move(input, moveSpeed);
            _materialScroller.ScrollMaterial(input);
        }
        
        if (shootInput)
        {
            _shooter.Shoot(transform.forward);
        }
    }
}
