📋 IEnemy.cs

📋 Interface 👾 Enemy 🤖 AI 

public interface IEnemy
{
    void Initialize(Vector3 spawnPosition);
    void Move();
    void Attack();
    void TakeDamage(float damage);
    void Die();
    void SetTarget(Transform target);
}

📝 Arayüz Açıklaması

  • 🎯 Temel Amaç: Tüm düşman tipleri için temel davranışları tanımlayan sözleşme
  • 🔧 Özellikler:
    • Temel düşman davranışları
    • Yaşam döngüsü yönetimi
    • Hareket ve saldırı kontrolü
    • Hedef yönetimi

🔄 Metodlar

  • Initialize: 
    void Initialize(Vector3 spawnPosition);
    • Düşmanı belirtilen pozisyonda başlatır
    • Başlangıç değerlerini ayarlar
  • Move: 
    void Move();
    • Düşmanın hareket mantığını yönetir
    • Hedefe doğru hareket sağlar
  • Attack: 
    void Attack();
    • Saldırı mantığını yönetir
    • Strateji desenini kullanır
  • TakeDamage: 
    void TakeDamage(float damage);
    • Hasar alma mantığını yönetir
    • Can değerini günceller
  • Die: 
    void Die();
    • Ölüm mantığını yönetir
    • Temizleme işlemlerini yapar
  • SetTarget: 
    void SetTarget(Transform target);
    • Hedef atama mantığını yönetir
    • Takip edilecek hedefi belirler

📝 Kullanım Örneği


public class EnemyManager : MonoBehaviour
{
    private List enemies = new List();
    private Transform playerTarget;

    public void SpawnEnemy(IEnemy enemy, Vector3 position)
    {
        // Düşmanı başlat
        enemy.Initialize(position);
        enemy.SetTarget(playerTarget);
        enemies.Add(enemy);
    }

    public void UpdateEnemies()
    {
        foreach (var enemy in enemies)
        {
            // Düşman davranışlarını güncelle
            enemy.Move();
            enemy.Attack();
        }
    }}
